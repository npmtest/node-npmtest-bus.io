var util = require('util')
  , events = require('events')
  , debug = require('debug')('bus.io-messages:messages')
  , common = require('bus.io-common')
  , Message = common.Message
  ;
 
module.exports = Messages;
 
/**
 * Produces and publishes socket messages
 */
 
function Messages () {
  var self = this;
 
  if (!(this instanceof Messages)) return new Messages();
 
  events.EventEmitter.call(this);
 
  debug('new messages');
 
  /**
   * When a socket is accepted
   *
   * @param {object} socket
   */
 
  this.onConnection = function (socket) {
 
    debug('on connection %s', socket.id);
 
    /**
     * This is used to add listeners to sockets when the Messages.prototype.action()
     * method is invoked
     *
     * @param {string} name
     */
 
    function onAction (name) {
      debug('on action %s', name);
      if (socket.listeners(name).indexOf(self.onMessage) < 0) {
        socket.on(name, function () {
          debug('socket on %s', name);
          var args = Array.prototype.slice.call(arguments);
          self.onMessage(socket, [name].concat(args));
        });
      }
    }
   
    self.on('action', onAction);
 
    /*
     * take the current actions and set up handlers for them
     */
 
    if (self.actions()) {
      self.actions().forEach(onAction);
    }
 
 
    /*
     * when we disconnect we want to remove the handlers and the onAction
     * method
     */
 
    socket.on('disconnect', function () {
      debug('on disconnect %s', socket.id);
      self.removeListener('action', onAction);
      self.actions().forEach(function (name) {
        debug('removing listeners for %s', name);
        socket.removeAllListeners(name);
        debug('listener count %s for %s', socket.listeners(name).length, name);
      });
    });
 
  };
 
  /*
   * Called when we receive data from the socket
   * It will build up the actor, target, and content
   * to produce a json object that will be published
   * onto an publisher / event emitter
   */
 
  this.onMessage = function (socket, args) {
    debug('on message %s', socket.id);
    args = args || [];
    if (!args.length) return;
    var message, isMessage = false;
    if (args.length === 2 && args[1] && args[1].isMessage) {
      isMessage = true;
      message = Message(args[1]);  
    }
    else {
      message = Message();
    }
    message.action(args.shift());
    self.actor(socket, function (err, actor) {
      debug('on actor %s, %s', socket.id, actor);
      if (err) {
        return self.emit('error', err, socket, args);
      }
      message.actor(actor);
      self.target(socket, args, function (err, target) {
        debug('on target %s, %s', socket.id, actor);
        if (err) {
          return self.emit('error', err, socket, args);
        }
        message.target(target);
        if (!isMessage) message.content(args);
        self.emit('message', message, socket);
      });
    });
  };
 
  /*
   * the middleware fucntion we attach to socket.io. It looks for events and if
   * we have auto propagation turned on it will trigger events that are *not*
   * in the current action set.
   */
 
  this.middleware = function (socket, next) {
    debug('middleware invoked on socket %s', socket.id);
    socket.onevent = self.onSocketEvent;
    next();
  };
 
  /*
   * The function we override the onevent with.
   */
 
  // TODO lets use socket.io-events
 
  this.onSocketEvent = function(packet) {
    debug('on socket event for socket %s', this.id);
    var args = packet.data || [];
 
    if (null != packet.id) {
      args.push(this.ack(packet.id));
    }
 
    if (this.listeners(args[0]).length) {
      events.EventEmitter.prototype.emit.apply(this, args);
    }
    else if (self.autoPropagate()) {
      self.onMessage(this, args);
    }
 
  };
 
  this.setMaxListeners(0);
}
 
util.inherits(Messages, events.EventEmitter);
 
/**

Messages.make = function () {
  debug('make');
  return new Messages();
};
 
/**
 * Creates a new instance and has it listen to "io"
 *
 * @api public
 * @param {object} io
 * @return Messages
 */
 
Messages.listen = function (io) {
  debug('listen');
  var messages = this.make();
  messages.attach(io);
  return messages;
};
 
/**
 * Attaches our #onConnection method to the io object
 *
 * @api public
 * @param {object} io
 * @return Messages
 */
 
Messages.prototype.attach = function (io) {
  debug('attach');
  if (io.sockets.fns.indexOf(this.middleware) < 0) {
    io.use(this.middleware);
  }
  io.on('connection', this.onConnection);
  return this;
};
 
/**
 * Either sets the actor query method or invokes it
 *
 * Set
 *
 * @api public
 * @param {Function} o
 *
 * Invoke
 *
 * @param {Object} o The socket
 * @param {Function} cb
 *
 * @return Messages
 */
 
Messages.prototype.actor = function (o, cb) {
  debug('actor');
  var type = typeof o;  
  if (arguments.length === 0) {
    if (!this._actor) {
      this._actor = function (socket, cb) {
        if (typeof cb === 'function') {
          cb(null, socket.id);
        }
      };
    }
    return this._actor;
  }
  else if (arguments.length === 1) {
    if (type === 'undefined' || !o) {
      return this;
    }
    if (type === 'function') {
      this._actor = o
    }
  }
  else if (arguments.length > 1) {
    if (type !== 'function') {
      if (!this._actor) {
        this._actor = function (socket, cb) {
          if (typeof cb === 'function') {
            cb(null, socket.id);
          }
        };
      }
      this._actor(o, cb);
    }
  }
  return this;
};
 
/**
 * Either sets the target query method or invokes it
 *
 * Set
 *
 * @api public
 * @param {Function} o
 *
 * Invoke
 *
 * @param {Object} o The socket
 * @param {Array} p The arguments emitted from the socket
 * @param {Function} cb
 *
 * @return Messages
 */
 
Messages.prototype.target = function (o, p, cb) {
  debug('target');
  var type = typeof o;  
  if (arguments.length === 0) {
    if (!this._target) {
      this._target = function (socket, params, cb) {
        if (typeof cb === 'function') {
          cb(null, socket.id);
        }
      };
    }
    return this._target;
  }
  else if (arguments.length === 1) {
    if (type === 'undefined' || !o) {
      return this;
    }
    if (type === 'function') {
      this._target = o
    }
  }
  else if (arguments.length > 2) {
    if (type !== 'function') {
      if (!this._target) {
        this._target = function (sock, params, cb) {
          if (typeof cb === 'function') {
            if (params[0] && params[0].isMessage) {
              cb(null, params[0].target || sock.id);
            }
            else {
              cb(null, sock.id);
            }
          }
        };
      }
      this._target(o, p, cb);
    }
  }
  return this;
};
 
/**
 * Triggers the binding of an event handler to all sockets for the given event name
 *
 * @api public
 * @param {string} name
 * @return Messages
 */
 
Messages.prototype.action = function (name) {
  if (typeof name === 'string' && this.actions().indexOf(name) < 0) {
    debug('add action %s', name);
    this.actions().push(name)
    this.emit('action', name);
  }
  return this;
};
 
/**
 * initializes the actions we have set so fare
 *
 * @api public
 * @return Array
 */
 
Messages.prototype.actions = function () {
  if (!this._actions) {
    this._actions = [];
  }
  return this._actions;
};
 
/**
 * Gets / Sets the state of auto propagation
 *
 * @api public
 * @return Boolean
 */
 
Messages.prototype.autoPropagate = function (v) {
  debug('autoPropagate %s', v);
  if (typeof v === 'boolean') {
    this._autoPropagate = v;
    return this;
  }
  if (typeof this._autoPropagate !== 'boolean') {
    this.autoPropagate(false);
  }
  return this._autoPropagate;
};

/**
 * The bus
 *
 * @param {object} io socket.io instance or http.Server
 */
 
function Server (a, b) {
 
  if (!(this instanceof Server)) {
    if (!arguments.length && server) return server;
    var instance = new Server(a, b);
    if (!server) server = instance;
    return instance;
  }
 
 
  debug('new server', a, b);
 
  var self = this;
 
  events.EventEmitter.call(this);
 
  /**
   * Called when we are supposed to publish the message. this is bound to a 
   * Builder instance an Receiver instances
   *
   * @api private
   * @param {Message} message
   */
 
  this.onPublish = function (message) {
    debug('on publish ', message.id());
    self.exchange().publish(message);
  };
 
 
  /**
   * Called when we receive a socket connection, this is bound to the
   * socket.io instance
   *
   * @api private
   * @param {Socket} socket
   */
 
  this.onConnection = function (socket) {
 
    debug('on connection %s', socket.id);
 
    function handle (message) {
      debug('from excahnge pubsub %s, %s', message.id(), socket.id);
      self.emit('from exchange pubsub', message, socket);
    }
 
    self.messages().actor(socket, function (err, actor) {
      if (err) return self.emit('error', err);
      // TODO this could be cleaned up using the alias and unalias methods
      self.exchange().subscribe(actor, handle, function (err, channel) {
        debug('subscribed to chanel %s with socket %s', actor, socket.id);
        if (err) return self.emit('error', err);
      });
      socket.on('disconnect', function () {
        self.exchange().unsubscribe(actor, handle, function (err, channel) {
        debug('unsubscribed from chanel %s with socket %s', actor, socket.id);
          if (err) return self.emit('error', err);
        });
      });
    });
 
  };
 
  /**
   * Called when we reeived a message from the Messages.  This is bound
   * to the Messages instance.  It will then emit an event for the
   * SocketReceiver instance to handle. This is when the Messages instance
   * finishes createing the message and before the message is passed along to
   * the Receiver for processing before dispatched to the Exchange.
   *
   * @api private
   * @param {object} message
   * @param {Socket} socket
   * @see Messages
   * @see Receiver
   */
 
  this.onMessage = function (message, socket) {
    debug('on message from socket %s, %s', (message && message.data ? message.data.id : null), (socket ? socket.id : null));
    if (message.data.action === '_flag') {
      debug('flag set');
      socket._flag = 1;
      return socket.emit('_flag', 1);
    }
    self.emit('from socket', message, socket);
  };
 
  /**
   * Called after we have received a message from the Exchange
   *
   * @api private
   * @param {Message} message
   * @param {Socket} socket
   * @see Exchange
   * @see Receiver
   */
 
  this.onReceivedPubSub = function (message, socket) {
    debug('sending message %s to socket %s', (message && message.data ? message.data.id : null), (socket ? socket.id : null));
    if (socket._flag) {
      debug('flag set for bus.io-client');
      socket.emit.apply(socket, [message.data.action, message]);
    }
    else {
      debug('socket.io client');
      socket.emit.apply(socket, [message.data.action, message.data.actor].concat(message.data.content).concat([message.data.target, message.data.created]));
    }
  };
 
  /**
   * Called after we have received a message from the socket.  This is after
   * the SocketMessage instance creates a message and the Receiver instance
   * processes it through the middleware.
   *
   * @api private
   * @param {Message} message
   * @param {Socket} socket
   * @see Messages
   * @see Receiver
   */
 
  this.onReceivedSocket = function (message, socket) {
    debug('received a message %s from the socket %s ', (message && message.data ? message.data.id : null), (socket ? socket.id : null));
    self.message(message).deliver();
  };
 
  /**
   * Called when we receveie a message on the Queue
   *
   * @api private
   * @param {object} message
   * @see Exchange
   */
 
  this.onReceivedQueue = function (message) {
    debug('received message %s from queue', message.id());
    self.emit('from exchange queue', message);
  };
 
  /**
   * Handles our error
   *
   * @api private
   * @param {mixed} err
   */
 
  this.onError = function () {
    debug('on error', arguments);
    console.error.apply(console,Array.prototype.slice.call(arguments));
  };
 
  this.addListener('error', this.onError);
 
  this.incomming();
 
  this.processing();
 
  this.outgoing();
 
  this.autoPropagate(true);
 
  if (a) {
    debug('we have something so we will try to listen to it');
    this.listen(a, b);
  }
}

/**
 * Listen on the port or socket.io instance
 *
 * @api public
 * @param {mixed} Either a number or a Socket.IO instance
 * @param {Object} Options
 * @return Server
 */
 
Server.prototype.listen = function (a, b) {
  var attach = function (server) {
    if (!server) {
      server = http.Server(function (req, res) {
        res.writeHead(404);
        res.end();
      });
    }
    var handlers = server.listeners('request').slice(0);
    server.removeAllListeners('request');
    server.on('request', function (req, res) {
      debug('on request %s', req.url);
      if (0 === req.url.indexOf('/bus.io/bus.io.js')) {
        try {
					var etag = req.headers['if-none-match'];
					if (etag) {
						if (clientVersion === etag) {
							res.writeHead(304);
							res.end();
							return;
						}
					}
					else {
						res.setHeader('Content-Type', 'application/javascript');
						res.setHeader('ETag', clientVersion);
						res.writeHead(200);
						res.end(clientSource);
					}
        }
        catch(e) {
          console.error(e);
          console.log(e.stack);
        }
        return;
      }
      else {
        debug('not what we are looking for');
        for (var i = 0; i < handlers.length; i++) {
          handlers[i].call(server, req, res);
        }
      }
    });
    return server;
  };
 
  if (a instanceof sio) {
    debug('a socket.io instance');
    console.warn('It is not recommended to pass in a Socket.IO Server instance.  If you need to manipulate the Socket.IO server instance please call bus.io() to get it');
    a.attach(attach(), b);
    this.io(a);
  }
  else if (!isNaN(a)) {
    debug('a port number %s', a);
    this.io(sio(attach().listen(Number(a)), b));
  }
  else if (a instanceof http.Server) {
    debug('an http server instnace');
    this.io(sio(attach(a), b));
  }
  else {
    console.error('must pass in either an http.Server(), or a Number');
  }
 
  return this;
};

/**
 * sets up an alias for the actor / socket
 *
 * @api public
 * @param {Socket} socket
 * @param {string} name
 * @param {Function} done * optional
 * @return Server
 */
 
Server.prototype.alias = function (socket, name, done) {
  debug('aliasing a socket %s to the name %s', socket.id, name);
 
  var self = this;
  var handle = function (message) {
    debug('[alias "%s"] from exchange pubsub message %s, socket %s', name, message.id(), socket.id);
    self.emit('from exchange pubsub', message, socket);
  };
  handle.id = socket.id;
 
  this.exchange().subscribe(name, handle, function (err, channel) {
 
    if (err) return self.emit('error', err);
 
    debug('[alias "%s"] subscribed  %s, socket %s', name, channel, socket.id);
 
    socket.on('disconnect', function () {
      self.exchange().unsubscribe(name, handle, function (err, channel) {
        if (err) return self.emit('error', err);
        debug('[alias "%s"] unsubscribed  %s, socket %s, listeners now %s', name, channel, socket.id, self.exchange().listeners('channel ' + name).length);
      });
    });
 
    if (typeof done === 'function') {
      done(err, channel);
    }
  });
 
  return this;
};
 
/**
 * removes an alias for the actor / socket
 *
 * @api public
 * @param {Socket} socket
 * @param {string} name
 * @param {Function} done * optional
 * @return Server
 */
 
Server.prototype.unalias = function (socket, name, done) {
  debug('aliasing a socket %s to the name %s', socket.id, name);
 
  var self = this;
  
  // TODO a hash would be better to store this information
  var listeners = this.exchange().listeners('channel '+ name), handle = null;
  debug('searching for the handler in these listeners %s', listeners.length);
  for (var i=0; i<listeners.length; i++) {
    if (listeners[i].id && listeners[i].id === socket.id) {
      handle = listeners[i]; 
      break;
    }
  }
 
  if (handle) {
    debug('found handler');
 
    debug('unsubscribing %s', name);
    this.exchange().unsubscribe(name, handle, function (err, channel) {
 
      if (err) return self.emit('error', err);
 
      debug('[unalias "%s"] unsubscribed  %s, socket %s, listeners now %s', name, channel, socket.id, self.exchange().listeners('channel ' + name).length);
 
      if (typeof done === 'function') {
        done(err, channel);
      }
 
    });
 
  }
  else {
    debug('did not find handler');
    if (typeof done === 'function') {
      done(null, name);
    }
  }
 
  return this;
};
